public class FileWriterEx {
public static void main(String[] args) throws IOException {
	FileWriter fp=new FileWriter("xyz.txt");
	fp.write("Hello!");
	fp.write("Welcome");
	fp.write("to");
	fp.write("Atcs");
	fp.write(23);
	fp.close();
	System.out.println("Writing in file using file writer");
}
}
Question 4

public class BufferWriterEx {
public static void main(String[] args) throws IOException {
	FileWriter fp=new FileWriter("xyz.txt");
	BufferedWriter bp=new BufferedWriter(fp);
	bp.write("Hello!");
	bp.write("Welcome");
	bp.write("to");
	bp.write("Atcs");
	bp.write(23);
	bp.write(56);
	bp.close();
	System.out.println("Writing in file using buffer writer");
}
}
